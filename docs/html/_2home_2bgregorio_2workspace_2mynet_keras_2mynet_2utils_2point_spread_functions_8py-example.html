35 def _normalize_kernel(kernel, mode='integral'):
36  """!@brief
37  Normalize the filter kernel.
38 
39  @param mode : {'integral', 'peak'}
40  One of the following modes:
41  * 'integral' (default)
42  Kernel is normalized such that its integral = 1.
43  * 'peak'
44  Kernel is normalized such that its peak = 1.
45  """
46 
47  if mode == 'integral':
48  normalization = kernel.sum()
49  elif mode == 'peak':
50  normalization = kernel.max()
51  else:
52  raise ValueError("invalid mode, must be 'integral' or 'peak'")
53 
54  # Warn the user for kernels that sum to zero
55  if normalization == 0:
56  print('The kernel cannot be normalized because it sums to zero.')
57  else:
58  np.divide(kernel, normalization, kernel)
59 
60  return kernel

127  @example
128  --------
129  Kernel response:
130 
131  .. plot::
132  :include-source:
133 
134  import matplotlib.pyplot as plt
135  from .utils import AiryDiskKernel
136  airydisk_2D_kernel = AiryDiskKernel(10)
137  plt.imshow(airydisk_2D_kernel, interpolation='none', origin='lower')
138  plt.xlabel('x [pixels]')
139  plt.ylabel('y [pixels]')
140  plt.colorbar()
141  plt.show()
142  """
143  # Define kernel size according to radius value
144  k_size = _round_up_to_odd_integer(3*radius)
145 
146  # Set ranges where to evaluate the model
147  if k_size % 2 == 0: # even kernel
148  xy_range = (-(int(k_size)) // 2 + 0.5, (int(k_size)) // 2 + 0.5)
149  else: # odd kernel
150  xy_range = (-(int(k_size) - 1) // 2, (int(k_size) - 1) // 2 + 1)
151 
152  x = np.arange(*xy_range)
153  y = np.arange(*xy_range)
154  x, y = np.meshgrid(x, y)
155 
156  amplitude = 1
157  x_0 = 0
158  y_0 = 0
159 
160  try:
161  from scipy.special import j1, jn_zeros
162  _rz = jn_zeros(1, 1)[0] / np.pi # equals to 1.22
163  _j1 = j1
164  except ValueError:
165  raise ImportError('AiryDisk2D model requires scipy > 0.11.')
166 
167  r = np.sqrt((x - x_0) ** 2 + (y - y_0) ** 2) / (radius / _rz)
168 
169  # Since r can be zero, we have to take care to treat that case
170  # separately so as not to raise a numpy warning
171  z = np.ones(r.shape)
172  rt = np.pi * r[r > 0]
173  z[r > 0] = (2.0 * _j1(rt) / rt) ** 2
174  z *= amplitude
175 
176  # Normalize kernel values
177  kernel = _normalize_kernel(z)

192 def MotionKernel(k_size=3,
193  k_angle=0,
194  k_type=None):
195  """!@brief
196  Create a kernel simulating the motion in the image by using random params.
197 
198  @param k_size : The kernel size.
199  @param k_angle : The angle of motion in the kernel.
200  @param k_type : The kernel type {'line', 'curve'}.
201 
202  @return
203  A kernel for motion simulation.
204  """
205 
206  # Initialize kernel with zeros
207  k = np.zeros((k_size,k_size))
208 
209  if k_type is None:
210  types = ['line', 'curve']
211  # Choose randomly the kernel type
212  k_type = str(np.random.choice(types))
213 
214  if k_type == 'line':
215  # Weighting the motion
216  s=1
217  x_w = np.linspace(lognorm.ppf(0.01, s), lognorm.ppf(0.99, s), k_size)
218  rv = lognorm(s)
219 
220  # plt.plot(x_w, rv.logpdf(x_w), 'k-', lw=2, label='frozen pdf')
221  # plt.show()
222 
223  line_1 = math.trunc(k_size/2)
224  k[line_1:line_1+1,:] = rv.logpdf(x_w)
225 
226  elif k_type == 'curve':
227  n_pts=100 # Number of points to create the function
228  x = np.linspace(-0.1, 0.05, n_pts)
229  y = np.cos(x)
230 
231  # Simulate the change of frequency in the kernel
232  # delta_x = k_size/4
233  delta_y = k_size/4
234 
235  # Discretizing function
236  x = minmax_scale(x,feature_range=(0,k_size-1)).astype(int)
237  y = minmax_scale(y,feature_range=(delta_y,k_size-delta_y)).astype(int)
238 
239  # plt.plot(x, y, 'k-', lw=2, label='frozen pdf')
240  # plt.show()
241 
242  # Weighting the motion
243  s=1
244  x_w = np.linspace(lognorm.ppf(0.01, s), lognorm.ppf(0.99, s), n_pts)
245  rv = lognorm(s)
246 
247  # plt.plot(x_w, rv.logpdf(x_w), 'k-', lw=2, label='frozen pdf')
248  # plt.show()
249 
250  k[y,x] = rv.logpdf(x_w)
251 
252  else:
253  raise ValueError("Invalid option for motion kernel.")
254 
255  # Blur kernel
256  # k = cv2.GaussianBlur(k,(5,3),cv2.BORDER_DEFAULT)
257 
258  # Rotate kernel
259  if k_angle:

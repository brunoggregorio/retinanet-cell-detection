195 def run(generator, args, anchor_params):
196  """!@brief
197  Main loop.
198 
199  @param generator : The generator to debug.
200  @param args : Parseargs args object.
201  """
202  # display images, one at a time
203  for i in range(generator.size()):
204  # load the data
205  image = generator.load_image(i)
206  annotations = generator.load_annotations(i)
207 
208  # Apply random transformations
209  # if args.random_transform or args.random_deformable or args.random_photometric:# or args.random_psf_blur:
210  if True:
211  image, annotations = generator.random_transform_group_entry(image, annotations)
212 
213  # resize the image and annotations
214  if args.resize:
215  image, image_scale = generator.resize_image(image)
216  annotations['bboxes'] *= image_scale
217 
218  anchors = anchors_for_shape(image.shape, anchor_params=anchor_params)
219  positive_indices, _, max_indices = compute_gt_annotations(anchors, annotations['bboxes'])
220 
221  # draw anchors on the image
222  if args.anchors:
223  draw_boxes(image, anchors[positive_indices], (255, 255, 0), thickness=1)
224 
225  # draw annotations on the image
226  if args.annotations:
227  # draw annotations in red
228  draw_annotations(image, annotations, color=(0, 0, 255), label_to_name=generator.label_to_name)
229 
230  # draw regressed anchors in green to override most red annotations
231  # result is that annotations without anchors are red, with anchors are green
232  draw_boxes(image, annotations['bboxes'][max_indices[positive_indices], :], (0, 255, 0))
233 
234  cv2.imshow('Image', image)
235  if cv2.waitKey() == ord('q'):
236  return False
237  return True
